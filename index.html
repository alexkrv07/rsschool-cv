          function deepCount(a) {
            if (a.length === 0 ) {
              return 0;
            }
            if (!Array.isArray(a)) {
              return 1;
            }
            return a.reduce((sum, item) => {

                return sum + 1 + deepCount(item);
              }
              return sum + deepCount(item);
            }, 0);
          }
